ASSEMBLY INTRODUCTION

Assembly language is a low-level programming language that provides a direct correspondence between the program and the machine code instructions of a specific CPU architecture.

machine_code.asm 
section .data
    msg db 'Hello, Cyber World!', 0xA
    len equ $ - msg

section .text
    global _start

_start:
    ; write message to stdout
    mov eax, 4      ; sys_write syscall
    mov ebx, 1      ; file descriptor (stdout)
    mov ecx, msg     ; message address
    mov edx, len     ; message length
    int 0x80          ; interrupt to invoke kernel

    ; exit program
    mov eax, 1      ; sys_exit syscall
    xor ebx, ebx    ; exit code 0
    int 0x80

Each assembly instruction corresponds directly to a machine code instruction that the CPU can execute. This provides maximum control over hardware resources.

WHY ASSEMBLY?

  • Maximum Performance: Direct hardware control eliminates overhead
  • Minimal Footprint: Tiny executable sizes (often kilobytes)
  • Hardware Access: Direct manipulation of CPU registers and memory
  • Reverse Engineering: Essential for malware analysis and security
  • Embedded Systems: Critical for microcontrollers with limited resources
  • Operating Systems: Bootloaders, kernel development, drivers
  • Game Development: Performance-critical game engine components
  • Cryptography: Optimized implementations of cryptographic algorithms

basic_operations.asm 
; Register operations
mov eax, 10     ; Load immediate value 10 into EAX
add eax, 5      ; EAX = EAX + 5  (now 15)
sub eax, 3      ; EAX = EAX - 3  (now 12)

; Memory operations
mov ebx, 0x1000  ; Load address
mov [ebx], eax  ; Store EAX at memory location 0x1000
mov ecx, [ebx]  ; Load from memory into ECX

; Bitwise operations
and eax, 0xF     ; Mask lower 4 bits
or eax, 0x10    ; Set bit 4
xor eax, eax    ; Clear EAX (fast zero)

; Control flow
cmp eax, 10     ; Compare EAX with 10
jg greater      ; Jump if greater
jl less         ; Jump if less
je equal        ; Jump if equal

x64_registers.asm 
; 64-bit extensions (x86-64)
; 64-bit: RAX, RBX, RCX, RDX, RSI, RDI, RBP, RSP
; Additional: R8-R15

; Using 64-bit registers
mov rax, 0xFFFFFFFFFFFFFFFF  ; Max 64-bit value
mov rbx, 0x123456789ABCDEF   ; Large constant

; Accessing smaller parts
; RAX (64) -> EAX (32) -> AX (16) -> AH/AL (8 high/low)
mov al, 0xFF        ; Set low byte of RAX
mov ah, 0x11        ; Set high byte of AX
mov eax, 0xDEADBEEF  ; Set lower 32 bits of RAX

; New registers in x64
mov r8, 100
mov r9, 200
add r8, r9         ; r8 = 300

common_instructions.asm 
; Data Transfer
mov eax, ebx    ; Copy EBX to EAX
xchg eax, ebx   ; Swap EAX and EBX
lea eax, [ebx+ecx*4]  ; Load effective address

; Arithmetic
add eax, 10     ; EAX = EAX + 10
sub ebx, ecx    ; EBX = EBX - ECX
inc eax         ; EAX++
dec ebx         ; EBX--
mul ecx         ; EDX:EAX = EAX * ECX
div ebx         ; EAX = EDX:EAX / EBX

; Bit Operations
and eax, 0xF     ; Bitwise AND
or eax, 0x10    ; Bitwise OR
xor eax, eax    ; Clear EAX
not eax         ; Bitwise NOT
shl eax, 2      ; Shift left by 2 (multiply by 4)
shr eax, 1      ; Shift right by 1 (divide by 2)

; Control Flow
jmp target      ; Unconditional jump
call function   ; Call subroutine
ret             ; Return from subroutine

; Stack Operations
push eax        ; Push EAX onto stack
pop ebx         ; Pop from stack into EBX
pushad          ; Push all general registers
popad           ; Pop all general registers

SYSTEM CALLS & INTERRUPTS

syscalls.asm 
; Linux x86 System Calls (int 0x80)
; EAX = syscall number, EBX, ECX, EDX = parameters

_start:
    ; write(1, message, length)
    mov eax, 4      ; sys_write = 4
    mov ebx, 1      ; fd = 1 (stdout)
    mov ecx, msg
    mov edx, len
    int 0x80

    ; exit(0)
    mov eax, 1      ; sys_exit = 1
    xor ebx, ebx    ; status = 0
    int 0x80

ARM ASSEMBLY

ARM (Advanced RISC Machine) is a reduced instruction set computing (RISC) architecture used in most mobile devices, embedded systems, and increasingly in servers and desktops.

arm_example.s 
/* ARM Assembly - Hello World */
.global _start

.section .data
message:
    .ascii "Hello, ARM World!\n"
len = . - message

.section .text
_start:
    /* write(stdout, message, len) */
    mov r0, #1      @ stdout file descriptor
    ldr r1, =message
    ldr r2, =len
    mov r7, #4      @ sys_write syscall number
    swi #0          @ software interrupt

    /* exit(0) */
    mov r0, #0      @ exit status
    mov r7, #1      @ sys_exit syscall number
    swi #0

ARM vs x86

ARM (RISC)

  • Reduced Instruction Set
  • Load/Store Architecture
  • Fixed-length Instructions
  • Energy Efficient
  • Many General Registers
  • Conditional Execution

x86 (CISC)

  • Complex Instruction Set
  • Memory-to-Memory Ops
  • Variable-length Instructions
  • High Performance
  • Fewer Registers
  • Backward Compatibility

ARM FEATURES

arm_features.s 
@ Conditional execution (unique to ARM)
cmp r0, #10
addgt r1, r1, #1   @ Only if r0 > 10
addle r1, r1, #2   @ Only if r0 <= 10

@ Barrel shifter - shift as part of instruction
add r0, r1, r2, LSL #2   @ r0 = r1 + (r2 << 2)
mov r0, r1, ROR #8      @ Rotate right by 8 bits

@ Load/Store multiple registers
stmfd sp!, {r0-r3, lr}   @ Push r0-r3 and lr
ldmfd sp!, {r0-r3, pc}   @ Pop r0-r3 and return

@ Thumb mode (16-bit instructions)
.thumb
adds r0, #1          @ 16-bit instruction
bx lr               @ Return from Thumb

@ NEON SIMD instructions
vadd.i32 q0, q1, q2   @ Vector addition
vmul.f32 q3, q4, q5   @ Vector multiplication

DEBUGGING ASSEMBLY

gdb_commands.txt 
(gdb) file program      # Load executable
(gdb) break _start      # Set breakpoint at _start
(gdb) run              # Start execution
(gdb) stepi            # Step one instruction
(gdb) info registers   # Show all registers
(gdb) print/x $eax      # Print EAX in hex
(gdb) x/10i $pc         # Examine 10 instructions at PC
(gdb) x/20xw $esp       # Examine 20 words at stack
(gdb) continue          # Continue execution
(gdb) quit             # Exit GDB
