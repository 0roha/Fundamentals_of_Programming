BASH INTRODUCTION

Bash (Bourne Again SHell) is a Unix shell and command language written by Brian Fox for the GNU Project. It's the default shell on most Linux distributions and macOS.

terminal
cyber@shell:~$ echo "Welcome to CyberShell"
Welcome to CyberShell

Bash combines features from the original Bourne shell (sh) with features from the Korn shell (ksh) and C shell (csh), making it a powerful tool for system administration and automation.

BASH SCRIPTING

Bash scripts automate tasks by executing commands in sequence. Every bash script starts with a shebang line:

hello.sh 
#!/bin/bash
# This is a simple bash script

echo "Hello, Cyber World!"
echo "Current date: $(date)"
echo "Current user: $USER"
echo "Hostname: $(hostname)"

# Make script executable with: chmod +x hello.sh
# Run with: ./hello.sh

Scripts can accept command-line arguments using positional parameters: $1, $2, $3, etc.

VARIABLES & CONTROL FLOW

control.sh 
#!/bin/bash
# Variables and control flow examples

# Variable assignment
name="CyberAdmin"
count=10

# If statement
if [ "$USER" = "root" ]; then
    echo "Running as root - proceed with caution!"
else
    echo "Running as user: $USER"
fi

# For loop
for i in {1..5}; do
    echo "Iteration $i"
done

# While loop
while [ $count -gt 0 ]; do
    echo "Countdown: $count"
    count=$((count-1))
done

# Case statement
case "$1" in
    start)
        echo "Starting service..."
        ;;
    stop)
        echo "Stopping service..."
        ;;
    *)
        echo "Usage: $0 {start|stop}"
        ;;
esac

FUNCTIONS & ERROR HANDLING

functions.sh 
#!/bin/bash
# Function examples

# Function definition
greet_user() {
    local username="$1"
    echo "Hello, $username! Welcome to the system."
}

# Function with return value
add_numbers() {
    local sum=$(($1 + $2))
    echo "$sum"
}

# Error handling
check_file() {
    if [ ! -f "$1" ]; then
        echo "Error: File $1 not found!" >&2
        return 1
    fi
    echo "File $1 exists."
    return 0
}

# Function calls
greet_user "Neo"
result=$(add_numbers 10 20)
echo "10 + 20 = $result"

# Check if file exists
check_file "/etc/passwd" || exit 1

# Exit codes
echo "Script completed with exit code: $?"

ADVANCED TECHNIQUES

advanced.sh 
#!/bin/bash
# Advanced bash techniques

# Arrays
files=(file1.txt file2.txt file3.txt)
echo "First file: ${files[0]}"
echo "All files: ${files[@]}"

# Associative arrays (bash 4+)
declare -A config
config[server]="cyberhost"
config[port]="8080"
echo "Server: ${config[server]}"

# Command substitution
current_date=$(date +%Y-%m-%d)
file_count=$(ls | wc -l)

# Process substitution
diff <(sort file1.txt) <(sort file2.txt)

# Trap signals
cleanup() {
    echo "Script interrupted. Cleaning up..."
    rm -f tempfile.txt
    exit 1
}

trap cleanup SIGINT SIGTERM

# Background processes
sleep 10 &
pid=$!
echo "Background process PID: $pid"

# Wait for process to complete
wait $pid
echo "Background process completed"

TEXT PROCESSING

text_processing.sh 
#!/bin/bash
# Advanced text processing with awk and sed

# Extract specific column from CSV
awk -F',' '{print $1, $3}' data.csv

# Sum values in a column
awk '{sum += $1} END {print "Total:", sum}' numbers.txt

# Find and replace in multiple files
sed -i 's/old_ip/new_ip/g' *.conf

# Extract lines between patterns
sed -n '/START/,/END/p' logfile.txt

# Remove empty lines
sed '/^$/d' file.txt

# Sort by column numerically
sort -t',' -k2 -n data.csv

# Find duplicate lines
sort file.txt | uniq -d

# Count unique values
cut -d',' -f1 data.csv | sort | uniq -c

# Join two files on a common field
join -t',' -1 1 -2 1 file1.csv file2.csv

CYBERSECURITY TOOLS

Bash is essential for cybersecurity professionals. Here are common tasks:

security_tasks.sh 
#!/bin/bash
# Security-related bash scripts

# Monitor failed login attempts
grep "Failed password" /var/log/auth.log | awk '{print $1, $2, $3, $11}' | sort | uniq -c

# Check for open ports
netstat -tuln | awk '/LISTEN/ {print $4}' | awk -F':' '{print $NF}' | sort -n

# Find files with SUID bit set (potential privilege escalation)
find / -type f -perm /4000 2>/dev/null

# Monitor network traffic
tcpdump -i eth0 -c 10

# Check for processes listening on ports
lsof -i -P -n | grep LISTEN

# Generate secure password
openssl rand -base64 12

# Check file integrity with checksum
sha256sum important_file.txt

# Scan for world-writable files
find / -type f -perm -o+w ! -path "/proc/*" ! -path "/sys/*" 2>/dev/null

AUTOMATION SCRIPTS

automation.sh 
#!/bin/bash
# System automation examples

# Backup script
backup_files() {
    local source_dir="$1"
    local dest_dir="$2"
    local timestamp=$(date +%Y%m%d_%H%M%S)
    local backup_name="backup_$timestamp.tar.gz"
    
    tar -czf "$dest_dir/$backup_name" -C "$source_dir" .
    
    if [ $? -eq 0 ]; then
        echo "Backup created: $dest_dir/$backup_name"
        return 0
    else
        echo "Backup failed!" >&2
        return 1
    fi
}

# Log analysis
analyze_logs() {
    local log_file="$1"
    
    echo "=== Top 10 IP addresses ==="
    awk '{print $1}' "$log_file" | sort | uniq -c | sort -nr | head -10
    
    echo "=== Top 10 requested pages ==="
    awk -F'"' '{print $2}' "$log_file" | awk '{print $2}' | sort | uniq -c | sort -nr | head -10
}

# System monitor
system_monitor() {
    echo "=== System Status ==="
    echo "Uptime: $(uptime -p)"
    echo "Load average: $(uptime | awk -F'load average:' '{print $2}')"
    echo "Memory usage: $(free -h | awk '/Mem:/ {print $3"/"$2}')"
    echo "Disk usage: $(df -h / | awk 'NR==2 {print $5}')"
    echo "Active users: $(who | wc -l)"
}
