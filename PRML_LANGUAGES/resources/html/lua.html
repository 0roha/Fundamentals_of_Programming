Variables & Data Types

local number = 42
local string = "Hello"
local boolean = true
local nil_value = nil
-- Table (arrays, objects, etc.)
local table = {1, 2, 3}

Lua is dynamically typed. Variables don't have types, only values do.

Control Structures

if age < 18 then
  print("Minor")
elseif age < 65 then
  print("Adult")
else
  print("Senior")
end

Lua features standard conditional statements and loops for program flow control.

Functions

function greet(name)
  return "Hello, " .. name
end

-- Anonymous function
local multiply = function(a, b)
  return a * b
end

Functions are first-class citizens in Lua and support multiple return values.

Tables

-- Array-like table
local fruits = {"apple", "banana", "cherry"}

-- Dictionary-like table
local person = {
  name = "John",
  age = 30,
  ["favorite color"] = "blue"
}

Tables are Lua's only data structure, used for arrays, objects, and more.

Loops

-- While loop
local i = 1
while i <= 5 do
  print(i)
  i = i + 1
end

-- For loop
for j = 1, 10, 2 do
  print(j)
end

Lua provides while, repeat-until, and for loops for iteration.

Metatables

local vector1 = {x = 1, y = 2}
local vector2 = {x = 3, y = 4}

local mt = {
  __add = function(a, b)
    return {x = a.x + b.x, y = a.y + b.y}
  end
}

setmetatable(vector1, mt)
setmetatable(vector2, mt)

local result = vector1 + vector2

Metatables allow you to define custom behavior for tables, including operator overloading.

Functions with Multiple Returns

function minmax(numbers)
  local min, max = math.maxinteger, math.mininteger
  for _, num in ipairs(numbers) do
    min = math.min(min, num)
    max = math.max(max, num)
  end
  return min, max
end

local lowest, highest = minmax({5, 2, 8, 1, 9})

Error Handling

local success, result = pcall(function()
  -- Code that might error
  if math.random() > 0.5 then
    error("Something went wrong!")
  end
  return "Success"
end)

if success then
  print(result)
else
  print("Error:", result)
end
