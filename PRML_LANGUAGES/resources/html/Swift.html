Swift Basics

Variables and Constants

Swift uses var for variables and let for constants.

var userName = "Neo" // Variable (mutable)
let maxLoginAttempts = 3 // Constant (immutable)
userName = "Trinity" // This is allowed
// maxLoginAttempts = 5 // This would cause an error

Data Types

Swift is a type-safe language with strong type inference.

var message: String = "Hello, System"
var accessLevel: Int = 9
var isOnline: Bool = true
var connectionSpeed: Double = 100.5

Control Flow

Swift provides standard control flow statements.

if accessLevel > 5 {
  print("Admin access granted")
} else {
  print("Insufficient privileges")
}

for i in 1...5 {
  print("Scanning sector \(i)")
}

var systemStatus = "online"
switch systemStatus {
  case "online":
    print("System operational")
  case "offline":
    print("System shutdown")
  default:
    print("Status unknown")
}

Functions & Collections

Functions

Functions are self-contained chunks of code that perform specific tasks.

// Function without parameters
func bootSystem() -> String {
  return "System boot sequence initiated"
}

// Function with parameters
func authenticateUser(username: String, password: String) -> Bool {
  return username == "admin" && password == "cyber"
}

// Function call
let authResult = authenticateUser(username: "admin", password: "cyber")
print("Authentication: \(authResult)")

Arrays and Dictionaries

Collections are used to store multiple values.

// Array
var systems = ["Security", "Network", "Database"]
systems.append("AI Core")
print("Total systems: \(systems.count)")

// Dictionary
var systemStatus = [
  "Security": "Online",
  "Network": "Offline",
  "Database": "Online"
]
systemStatus["AI Core"] = "Booting"

Optionals

Optionals handle the absence of a value.

var encryptedMessage: String? = "Secret Data"
encryptedMessage = nil // No value

// Optional binding
if let message = encryptedMessage {
  print("Decrypted: \(message)")
} else {
  print("No message to decrypt")
}

// Nil coalescing
let safeMessage = encryptedMessage ?? "Default Message"
